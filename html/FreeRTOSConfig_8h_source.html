114 /* The lowest interrupt priority that can be used in a call to a "set priority"
115  * function. */
116 #define configLIBRARY_LOWEST_INTERRUPT_PRIORITY 0x7
117 
118 /* The highest interrupt priority that can be used by any interrupt service
119  * routine that makes calls to interrupt safe FreeRTOS API functions. DO NOT CALL
120  * INTERRUPT SAFE FREERTOS API FUNCTIONS FROM ANY INTERRUPT THAT HAS A HIGHER
121  * PRIORITY THAN THIS! (higher priorities are lower numeric values. */
122 #define configLIBRARY_MAX_SYSCALL_INTERRUPT_PRIORITY 5
123 
124 /* Interrupt priorities used by the kernel port layer itself. These are generic
125 * to all Cortex-M ports, and do not rely on any particular library functions. */
126 #define configKERNEL_INTERRUPT_PRIORITY \
127  ( configLIBRARY_LOWEST_INTERRUPT_PRIORITY << ( 8 - configPRIO_BITS ) )
128 
129 /* !!!! configMAX_SYSCALL_INTERRUPT_PRIORITY must not be set to zero !!!!
130  * See http://www.FreeRTOS.org/RTOS-Cortex-M3-M4.html. */
131 #define configMAX_SYSCALL_INTERRUPT_PRIORITY \
132  ( configLIBRARY_MAX_SYSCALL_INTERRUPT_PRIORITY << ( 8 - configPRIO_BITS ) )
